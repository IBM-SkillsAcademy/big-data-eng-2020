# Run with root account
awk -v RS='\r?\n' ' BEGIN { FS = OFS = "," } 
    { 
      split($2, email, /@/);
      print "useradd -m -g hdfs "$1; 
      print "echo \"" email[1] "\" | passwd --stdin "$1;
      print "hadoop fs -mkdir /user/"$1;
      print "hadoop fs -chown "$1":hadoop /user/"$1;
      print "hadoop fs -chmod a-x /user/"$1;
      print "hadoop fs -chmod g-x /user/"$1;
    }
' users1.csv > create_studentAccounts
